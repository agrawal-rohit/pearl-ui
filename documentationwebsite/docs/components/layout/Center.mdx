Center is a layout component that centers its child within itself, given `width` and `height`. By default, it renders a [Box](./box) element.

## Import

```jsx
import { Center } from "pearl-ui";
```

## Usage

```jsx
<Center height={100} backgroundColor="neutral.300">
  This is the screen
</Center>
```

## Example

import Snack from "../../../src/components/ExpoSnack";
import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";

<Tabs
  defaultValue="light"
  values={[
    { label: 'Light', value: 'light', },
    { label: 'Dark', value: 'dark', },
  ]
}>
<TabItem value="light">

<Snack snackId="@agrawal-rohit/screen-light" />

</TabItem>

<TabItem value="dark">

<Snack snackId="@agrawal-rohit/screen-dark" />

</TabItem>
</Tabs>
